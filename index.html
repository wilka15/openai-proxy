import express from "express"; import cors from "cors"; import fetch from "node-fetch"; const app = express(); app.use(cors()); app.use(express.json()); const PORT = process.env.PORT || 3000; const OPENAI_KEY = process.env.OPENAI_API_KEY; app.get("/", (req, res) => { res.send("✔ OpenAI Proxy is running"); }); app.post("/v1/responses", async (req, res) => { try { const response = await fetch("https://api.openai.com/v1/responses", { method: "POST", headers: { "Content-Type": "application/json", "Authorization": `Bearer ${OPENAI_KEY}` }, body: JSON.stringify(req.body) }); const data = await response.json(); res.json(data); } catch (err) { res.status(500).json({ error: "Ошибка соединения с OpenAI" }); } }); app.listen(PORT, () => console.log(`Proxy running on port ${PORT}`));
